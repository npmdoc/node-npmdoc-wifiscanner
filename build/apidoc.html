function wifiscanner (options)

  • description and source-code
    function scanner(options) {
    options = options || {}
    const platform = platformSelect(options);
    nconf.file(`${__dirname}/config/${platform}.json`);

    let parser;
    switch(platform) {
        case "linux":
            parser = linuxparser;
        break;
        case "darwin":
            parser = darwinparser;
        break;
        case "win32":
            parser = windowsparser;
        break;
    }

    const scannerOptions = {
        binaryPath: options.binaryPath || nconf.get("binaryPath"),
        args: options.args || nconf.get("args"),
    }
    return new WiFiScanner(scannerOptions, parser);
}

    function extractMac(line) {
    const macPattern = /([0-9A-F]{2}[:-]){5}([0-9A-F]{2})/ig;
    return macPattern.exec(line.toLowerCase())[0];
